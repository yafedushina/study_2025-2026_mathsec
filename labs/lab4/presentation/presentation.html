def euclid(a: int, b: int) -> int: while a != 0 and b != 0: if a >= b: a %= b else: b %= a return a or b

euclid(a, b)

ef euclid_bin(a: int, b: int) -> int: g = 1 while a % 2 == 0 and b % 2 == 0: a /= 2 b /= 2 g = 2 u, v = a, b while u != 0: if u % 2 == 0: u /= 2 if v%2 == 0: v /= 2 if u>=v: u -= v else: v -= u return gv

Расширенный алгоритм Евклида

def euclid_ext(a: int, b: int) -> tuple[int]: if a == 0: return (b, 0, 1) else: div, x, y = euclid_ext(b % a, a) return (div, y - ( b // a ) * x, x)
